    deptno INT PRIMARY KEY,
    dept_name VARCHAR(50),
    location VARCHAR(50)
);

Employee Table
CREATE TABLE emp (
    emp_no INT PRIMARY KEY,
    birth_date DATE NOT NULL,
    first_name VARCHAR(14) NOT NULL,
    last_name VARCHAR(16) NOT NULL,
    gender CHAR(1) CHECK (gender IN ('M','F')) NOT NULL,
    hire_date DATE NOT NULL,
    salary FLOAT DEFAULT 7850.00,
    job VARCHAR(20) NOT NULL,
    deptno INT,
    manager_id INT,
    FOREIGN KEY (deptno) REFERENCES departments(deptno)
);

1.Display all employee names that starts with "a" and are at least 3 characters length
SQL> select First_name from emp where lower(first_name) like 'a%'and length(first_name) >= 3;


2.Display departments having Dollas, New York and Chicago locations
SQL> select name, location from departments where lower(location) = 'dallas' or lower(location)='new york' or lower(location) ='chicago';
OR 
select name, location from departments where lower(location) in ('dallas','new york','chicago');


3. Display the names of employees who are working as clerk, salesman or analyst and drawing a salary more than 3000.
SQL> select first_name, job, salary from emp where salary>=3000 and lower(job) in ('clerk','salesman','analyst');

4.Display the names of employees who are not working as managers.
SQL> select first_name from emp where lower(job) != 'manager';


5.Display the name of emp who earns highest salary
select first_name, salary from emp where salary=(select max(salary) from emp);

6.Display the employee number and name of employee working as CLERK and earning highest salary among CLERKS.
